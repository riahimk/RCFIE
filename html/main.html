parameters of the obstacle scattering problem

myScattering_problem.TYPE     =['Dirichlet ';'Neumann   ';'Impedance ';'GImpedance'];% padded
myScattering_problem.eta      =.5;    % coupling parameter
myScattering_problem.d        =[1; 0];
myScattering_problem.k        = 10;    % k: wave number of the scattering problem
%- - - - - - - - - - - - - - - - - - - - - - - -
myScattering_problem.T        = 2*pi/myScattering_problem.k;
myScattering_problem.n        = 120;%myScattering_problem.T/10;   % \pi/n discretization points on the boundary $\Gamma$
myScattering_problem.kk       = myScattering_problem.k^2;
myScattering_problem.type     = myScattering_problem.TYPE(1,:);
